# 114 Proyecto 2 - Wordpress con Mysql

## Configuración de Mysql

```yaml
mysql-service:
    image: mysql
    environment:
        MYSQL_ROOT_PASSWORD: wordpress$123
        MYSQ_DATABASE: wordpress
    volumes:
        - mysql_wordpress:/var/lib/mysql
    networks:
        - "wordpress_net"
```

## Configuración de Wordpress

```yaml
wordpress-service:
    image: wordpress
    environment:
        WORDPRESS_DB_HOST: mysql-service:3306
        WORDPRESS_DB_USER: root
        WORDPRESS_DB_PASSWORD: wordpress$123
        WORDPRESS_DB_NAME: wordpress
        WORDPRESS_TABLE_PREFIX: wp_
    networks:
        - "wordpress_net"
    ports:
        - "80:80"
```

## Configuración en Docker Compose

```yaml
services:
    mysql-service:
        image: mysql
        environment:
            MYSQL_ROOT_PASSWORD: wordpress$123
            MYSQ_DATABASE: wordpress
        volumes:
            - mysql_wordpress:/var/lib/mysql
        networks:
            wordpress_net:
                ipv4_address: "172.50.1.4"
    wordpress-service:
        depends_on: 
            - mysql-service
        image: wordpress
        environment:
            WORDPRESS_DB_HOST: 172.50.1.4:3306
            WORDPRESS_DB_USER: root
            WORDPRESS_DB_PASSWORD: wordpress$123
            WORDPRESS_DB_NAME: wordpress
            WORDPRESS_TABLE_PREFIX: wp_
        networks:
            - "wordpress_net"
        ports:
            - "80:80"

volumes:
    mysql_wordpress: {}

networks:
    wordpress_net:
        ipam:
            driver: bridge
            config:
                - subnet: "172.50.0.0/16"
                  ip_range: "172.50.1.0/24"
```
